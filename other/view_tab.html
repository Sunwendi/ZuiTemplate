引入文件： 
<link href="../Zui/lib/tabs/zui.tabs.min.css">
<script src="../Zui/lib/tabs/zui.tabs.min.js"></script>
html： 
<div class="tabs" id="tabsExample" style="height: 100%;"></div>
js: 
 var tabs = [{           定义标签页
    title: 'iframe 例子',
    url: 'GridManager.html',
    type: 'iframe',
    forbidClose: true
}, {
    title: '自定义例子',
    icon: 'icon-star',
    type: 'custom',
    content: function() {
        return '<div class="alert alert-block alert-success"><p>这里的内容是通过函数动态生成的，每次刷新下面的时间都会更新。</p><p>' + (new Date().format('yyyy-MM-dd hh:mm:ss')) +   '</p></div>';
    }
}];
$('#tabsExample').tabs({tabs: tabs});        初始化标签页管理器
