$(document).on("click",".judge_rows",function(){			获取当前已选中的行数据项
	var myDataGrid = $('#myDataGrid').data('zui.datagrid');			获取数据表格实例
	var selectedItems = myDataGrid.getCheckItems();				获取当前已选中的行数据项
	console.log(selectedItems)
})
$(document).on("click",".judge_news",function(){
	var myDataGrid = $('#myDataGrid').data('zui.datagrid');			获取数据表格实例
	myDataGrid.showMessage('这是一条提示消息', 'success');			在界面上显示 '这是一条提示消息' 消息。
})
